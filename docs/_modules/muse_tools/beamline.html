def psi_quad_from_transport(L_m, B_kG, a_cm, name, fringe_type="QXA"):
    a_m = a_cm * 0.01
    B_T = B_kG * 0.1
    G = B_T / a_m
    k1 = G / BRHO

    PSI_FRINGE = {
        "default": (0.209, -0.0768, 0.0552),
        "QSK":     (0.125, -0.0163, 0.0389),
        "QXA":     (0.190, -0.0125, 0.0777),
        "QTC":     (0.1667, 0.0055, 0.0894)
    }
    I1, I2, I3 = PSI_FRINGE.get(fringe_type, PSI_FRINGE["default"])

    L_eff = L_m + 2 * I1 * a_m
    quad_core = xt.Quadrupole(length=L_eff, k1=k1)

    dk1 = (I2 / a_m) * k1
    fringe_linear = xt.Multipole(knl=[0, dk1], length=0)

    k2 = (I3 / a_m**2) * k1
    fringe_nl = xt.Multipole(knl=[0, 0, k2], length=0)

    lim = xt.LimitRect(min_x=-a_m, max_x=a_m, min_y=-a_m, max_y=a_m)

    return [
        (f"{name}_fringe_in", fringe_linear),
        (f"{name}_fringe_nl_in", fringe_nl),
        (name, quad_core),
        (f"{name}_fringe_nl_out", fringe_nl.copy()),
        (f"{name}_fringe_out", fringe_linear.copy()),
        (f"lim_{name}", lim),
    ]

def quad_from_transport(L_m, B_kG, a_cm, name):
    a_m = a_cm * 0.01
    B_T = B_kG * 0.1
    G = B_T / a_m
    k1 = G / BRHO

    q = xt.Quadrupole(length=L_m, k1=k1)
    lim = xt.LimitRect(min_x=-a_m, max_x=a_m, min_y=-a_m, max_y=a_m)

    return [(name, q), (f"lim_{name}", lim)]

def dipole_from_transport(L_m, B0_kG, edge_entry_deg=0.0, edge_exit_deg=0.0, name=None):
    B0_T = 0.1 * B0_kG
    rho = P0_GeV / (0.299792458 * B0_T)
    h = 1.0 / rho
    angle = L_m * h

    bend = xt.Bend(length=L_m, angle=angle, k0=h)
    if name:
        bend.name = name

    edge_in = xt.DipoleEdge(h=h, e1=np.deg2rad(edge_entry_deg)) if edge_entry_deg else None
    edge_out = xt.DipoleEdge(h=h, e1=np.deg2rad(edge_exit_deg)) if edge_exit_deg else None

    return edge_in, bend, edge_out

def build_muse_line(
        psi_fringe=True,
        species="pi+",
        p0c=None,
        lim_fs11=[[-1.098e-2,1.098e-2],[-5.4e-2,5.4e-2]],
        lim_ifp=[[-0.105, 0.105],[-0.105, 0.105]],
        qfactors=[1.,1.,1.],
    ):

    if p0c is None:
        raise ValueError("build_muse_line requires p0c from main.py")

    from muse_tools.particles import make_reference_particle

    elements_lin, names_lin = [], []
    elements_full, names_full = [], []
    drift_count = 0
    limit_count = 0

    def fs11_slit(lim_fs11):
        return xt.LimitRect(
            min_x=lim_fs11[0][0],
            max_x=lim_fs11[0][1],
            min_y=lim_fs11[1][0],
            max_y=lim_fs11[1][1],
        )

    def ifp_slit(lim_ifp):
        return xt.LimitRect(
            min_x=lim_ifp[0][0],
            max_x=lim_ifp[0][1],
            min_y=lim_ifp[1][0],
            max_y=lim_ifp[1][1],
        )
    

    def add_drift(L_m, name=None):
        nonlocal drift_count
        if not name:
            drift_count += 1
            name = f"drift_{drift_count}"

        d = drift(L_m)
        elements_lin.append(d)
        names_lin.append(name)
        elements_full.append(d)
        names_full.append(name)

    def add_quad(L_m, B_kG, a_cm, name, fringe_type="QXA"):
        if psi_fringe:
            elements = psi_quad_from_transport(L_m, B_kG, a_cm, name, fringe_type)
        else:
            elements = quad_from_transport(L_m, B_kG, a_cm, name)
        for ename, elem in elements:
            elements_full.append(elem)
            names_full.append(ename)
            if isinstance(elem, (xt.Quadrupole, xt.Multipole)):
                elements_lin.append(elem)
                names_lin.append(ename)

    def add_dipole(L_m, B0_kG, name, edge_entry_deg=0.0, edge_exit_deg=0.0):
        edge_in, bend, edge_out = dipole_from_transport(
            L_m, B0_kG, edge_entry_deg, edge_exit_deg, name
        )
        for elem, ename in zip(
            [edge_in, bend, edge_out],
            [name+"_edge_in", name, name+"_edge_out"]
        ):
            if elem:
                elements_lin.append(elem)
                elements_full.append(elem)
                names_lin.append(ename)
                names_full.append(ename)

    def add_limit(lim, name=None):
        nonlocal limit_count
        limit_count += 1
        lname = name if name else f"lim_{limit_count}"
        elements_full.append(lim)
        names_full.append(lname)

        
    # ------------------------------------------------------------
    # FULL SEQUENCE (unchanged)
    # ------------------------------------------------------------
    add_drift(0.835); add_drift(0.123)
    add_quad(0.324, -1.019, 7.5, "QTA11")
    add_drift(0.078); add_drift(0.053)
    add_quad(0.475, -1.089, 11.5, "QTB11")
    add_drift(0.110)
    add_quad(0.475, 1.221, 11.5, "QTB12")
    add_drift(1.0); add_drift(1.0); add_drift(1.0); add_drift(1.0); add_drift(0.13)
    add_drift(0.001*54); add_drift(0.004); add_limit(fs11_slit(lim_fs11), name="FS11"); add_drift(0.001*54)
    add_drift(0.239); add_drift(0.368)

    add_dipole(0.77539, 4.4432, "ASM11a", edge_entry_deg=27.9, edge_exit_deg=0.0)
    add_dipole(0.77539, 4.4432, "ASM11b", edge_entry_deg=0.0, edge_exit_deg=27.9)

    add_drift(0.408); add_drift(0.410); add_drift(0.539)
    add_quad(0.59, -1.2767, 12.0, "QSL11")
    add_drift(0.112)
    add_quad(0.59, 0.9166, 12.0, "QSL12")
    add_drift(1.043); add_limit(ifp_slit(lim_ifp), name="IFP"); add_drift(0.001*50); add_drift(0.437)

    add_quad(0.59, 0.9068, 12.0, "QSL13")
    add_drift(0.112)
    add_quad(0.59, -1.2763, 12.0, "QSL14")
    add_drift(0.903); add_drift(0.408)

    add_dipole(0.77539, 4.3508, "ASM12a", edge_entry_deg=27.9, edge_exit_deg=0.0)
    add_dipole(0.77539, 4.3508, "ASM12b", edge_entry_deg=0.0, edge_exit_deg=27.9)

    add_drift(0.408); add_drift(0.276)
    add_quad(0.59, -0.3865, 12.0, "QSL15")
    add_drift(0.112)
    add_quad(0.59, 0.5962*qfactors[2], 12.0, "QSL16")
    add_drift(1.038)
    add_quad(0.59, -1.3454*qfactors[1], 12.0, "QSL17")
    add_drift(0.163)
    add_quad(0.59, 1.2568*qfactors[0], 12.0, "QSL18")
    add_drift(1.200); add_drift(0.336)
    add_drift(0.001, 'target')

    # ------------------------------------------------------------
    # Build lines
    # ------------------------------------------------------------
    line_lin = xt.Line(elements=elements_lin, element_names=names_lin)
    line_full = xt.Line(elements=elements_full, element_names=names_full)

    # ------------------------------------------------------------
    # Set reference particle (centralized species + momentum)
    # ------------------------------------------------------------
    ref = make_reference_particle(species, p0c)

    line_lin.particle_ref = ref
    line_full.particle_ref = ref

    line_full.build_tracker()

    return line_lin, line_full

def get_element_positions(line):
    positions = []
    s_positions = line.get_s_elements()
    for name, elem, s in zip(line.element_names, line.elements, s_positions):
        L = getattr(elem, "length", 0.0)
        etype = type(elem).__name__
        positions.append((name, etype, s, L))
    return positions

def element_color(etype):
    if "Quadrupole" in etype: return "royalblue"
    if "Multipole" in etype: return "red"
    if "LimitRect" in etype: return "green"
    return "gray"
