import xpart as xp

# Masses in eV/c^2, PDG IDs from PDG2024
PARTICLE_DATA = {
    "pi+":     {"mass": 139.57039e6,   "pdg_id": 211},
    "pi-":     {"mass": 139.57039e6,   "pdg_id": -211},
    "mu+":     {"mass": 105.6583755e6, "pdg_id": -13},
    "mu-":     {"mass": 105.6583755e6, "pdg_id": 13},
    "proton":  {"mass": 938.27208816e6,"pdg_id": 2212},
    "electron":{"mass": 510.998950e3,  "pdg_id": 11},
    "positron":{"mass": 510.998950e3,  "pdg_id": -11},
}

def make_reference_particle(species: str, p0c: float) -> xp.Particles:
    data = PARTICLE_DATA[species]
    return xp.Particles(
        mass=data["mass"],
        p0c=p0c,
        pdg_id=data["pdg_id"],
    )
