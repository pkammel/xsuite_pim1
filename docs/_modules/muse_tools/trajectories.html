def sigma0_from_beam_xy():
    """
    Initial beam definition from BEAM card.

    Returns:
        Sigma_x0, Sigma_y0: 2x2 covariance matrices
        phase_space: dict of half-widths for particle generation
    """
    
    sig_x  = 0.15e-2      # m
    sig_xp = 30e-3        # rad
    sig_y  = 0.15e-2      # m
    sig_yp = 120e-3       # rad

    Sigma_x0 = np.array([[sig_x**2,   0.0],
                         [0.0,        sig_xp**2]])

    Sigma_y0 = np.array([[sig_y**2,   0.0],
                         [0.0,        sig_yp**2]])

    phase_space = {
        "x_max":   sig_x,
        "px_max":  sig_xp,
        "y_max":   sig_y,
        "py_max":  sig_yp,
        "zeta_max": 0.0,
        "delta_max": 0.0,
    }

    return Sigma_x0, Sigma_y0, phase_space

def generate_particles_from_cfg(cfg, n_part=5000, scale=1.0):
    """
    Generate particles using BEAM-card half-widths stored in cfg.
    Species comes from cfg['species'], set in main.py.
    """
    ps = cfg["initial_phase_space"]
    line = cfg["line"]
    species = cfg["species"]  # <-- main.py controls this

    pdg_id = PARTICLE_DATA[species]["pdg_id"]

    x  = np.random.uniform(-scale*ps["x_max"],  scale*ps["x_max"],  n_part)
    px = np.random.uniform(-scale*ps["px_max"], scale*ps["px_max"], n_part)
    y  = np.random.uniform(-scale*ps["y_max"],  scale*ps["y_max"],  n_part)
    py = np.random.uniform(-scale*ps["py_max"], scale*ps["py_max"], n_part)

    part = line.build_particles(
        x=x, px=px, y=y, py=py,
        pdg_id=pdg_id
    )

    return part

def track_xy_trajectories(line, particles):
    n_elem = len(line.elements)
    n_part = particles.x.size

    x_traj = np.zeros((n_elem + 1, n_part))
    y_traj = np.zeros((n_elem + 1, n_part))

    x_traj[0, :] = particles.x.copy()
    y_traj[0, :] = particles.y.copy()

    for i, elem in enumerate(line.elements):
        elem.track(particles)

        # Mask lost particles immediately
        lost = particles.state == 0
        if np.any(lost):
            particles.x[lost] = np.nan
            particles.y[lost] = np.nan

        x_traj[i+1, :] = particles.x.copy()
        y_traj[i+1, :] = particles.y.copy()

    s_positions = line.get_s_elements()
    s_positions = np.append(s_positions, line.get_length())

    return s_positions, x_traj, y_traj

def get_element_positions(line):
    """
    Return list of (name, type, s_start, length) for real elements.
    """
    positions = []
    s = 0.0

    for name in line.element_names:
        elem = line.element_dict[name]
        L = getattr(elem, "length", 0.0) or 0.0

        etype = elem.__class__.__name__
        if etype.startswith("Quadrupole"):
            etype = "Quadrupole"
        elif etype.startswith("Bend") or etype.startswith("SBend"):
            etype = "Bend"
        elif etype.startswith("Multipole"):
            etype = "Multipole"
        elif etype.startswith("Drift"):
            etype = "Drift"
        else:
            if L == 0:
                continue

        positions.append((name, etype, s, L))
        s += L

    return positions

def plot_xy_trajectories(
        line, s, x_traj, y_traj,
        start_event=0,
        max_particles=50,
        xlim=None, ylim=None, slim=None
    ):
    """
    PSI-style x(s) and y(s) trajectory plot with shading and limits.
    """

    # particle selection
    n_part = x_traj.shape[1]
    n_plot = min(max_particles, n_part)

    # s-range cropping
    if slim is not None:
        smin, smax = slim
        mask = (s >= smin) & (s <= smax)
        s_plot = s[mask]
        x_plot = x_traj[mask, :n_plot]
        y_plot = y_traj[mask, :n_plot]
    else:
        s_plot = s
        x_plot = x_traj[:, :n_plot]
        y_plot = y_traj[:, :n_plot]

    fig, (ax_x, ax_y) = plt.subplots(
        2, 1, figsize=(14, 10), sharex=True, constrained_layout=True
    )

    # X trajectories
    for j in range(n_plot):
        ax_x.plot(s_plot, x_plot[:, j], color="darkorange", alpha=0.6)
    ax_x.set_ylabel("x [m]")
    ax_x.grid(True)

    # Y trajectories
    for j in range(n_plot):
        ax_y.plot(s_plot, y_plot[:, j], color="green", alpha=0.6)
    ax_y.set_xlabel("s [m]")
    ax_y.set_ylabel("y [m]")
    ax_y.grid(True)

    # axis limits
    if xlim is not None:
        ax_x.set_ylim(xlim)
    if ylim is not None:
        ax_y.set_ylim(ylim)

    # shading
    positions = get_element_positions(line)

    def color_fn(etype):
        if etype == "Quadrupole": return "red"
        if etype == "Bend": return "blue"
        if etype == "Multipole": return "purple"
        return "gray"

    for ax in (ax_x, ax_y):
        ymin, ymax = ax.get_ylim()
        full_height = ymax - ymin

        for name, etype, spos, L in positions:
            if L > 0 and etype in ("Quadrupole", "Bend", "Multipole"):
                if slim is None or (spos + L >= slim[0] and spos <= slim[1]):
                    ax.add_patch(Rectangle(
                        (spos, ymin), L, full_height,
                        color=color_fn(etype), alpha=0.15, zorder=0
                    ))
                    ax.text(
                        spos + L/2, ymin + 0.95*full_height,
                        name, rotation=90, fontsize=7,
                        ha="center", va="top"
                    )

    return fig
