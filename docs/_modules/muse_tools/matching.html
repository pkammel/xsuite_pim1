def measure_sigmas_from_matrices(line, Sigma_x0, Sigma_y0, target_name):
    # Build cumulative 2x2 matrices along the line
    Mx_list, My_list = get_cumulative_2x2_matrices(line, line.particle_ref)

    # Find index of target in the line
    idx = line.element_names.index(target_name)

    Mx = Mx_list[idx]
    My = My_list[idx]

    # Propagate covariance matrices: Sigma(s) = M Sigma0 M^T
    Sigma_x = Mx @ Sigma_x0 @ Mx.T
    Sigma_y = My @ Sigma_y0 @ My.T

    sigx = np.sqrt(Sigma_x[0, 0])
    sigy = np.sqrt(Sigma_y[0, 0])
    cost = sigx**2 + sigy**2
    return sigx, sigy

def scan_single_quad_matrix(line, quad_name, k1_range, n_steps,
                            target_name, Sigma_x0, Sigma_y0):

    k1_nom = line.element_dict[quad_name].k1
    k1_vals = np.linspace(k1_nom - k1_range, k1_nom + k1_range, n_steps)

    best = None

    for k1 in k1_vals:
        line.element_dict[quad_name].k1 = k1
        sigx, sigy, cost = measure_sigmas_from_matrices(
            line, Sigma_x0, Sigma_y0, target_name
        )

        if best is None or cost < best["cost"]:
            best = dict(k1=k1, sigx=sigx, sigy=sigy, cost=cost)

    line.element_dict[quad_name].k1 = best["k1"]
    return best

def coordinate_descent_triplet_matrix(
    line_optics,
    Sigma_x0,
    Sigma_y0,
    quad_names,
    k1_init,
    scan_ranges,
    n_steps,
    target="target",
    shrink_factor=0.8,
    n_iterations=1,
    min_range=1e-4,
):
    # store original k1 values for reporting
    k1_initial = np.array(k1_init, dtype=float)
    k1 = k1_initial.copy()
    scan_ranges = np.array(scan_ranges, dtype=float)

    def cost_function(line, Sigma_x0, Sigma_y0, target):
        sigx, sigy = measure_sigmas_from_matrices(line, Sigma_x0, Sigma_y0, target)
        return sigx**2 + sigy**2, sigx, sigy

    # store per-quad results
    results = {}

    for it in range(n_iterations):
        print(f"=== Iteration {it+1}/{n_iterations} ===")
        print(f"Current k1: {k1}")

        for i, qname in enumerate(quad_names):

            k1_center = k1[i]
            k1_min = k1_center - scan_ranges[i]
            k1_max = k1_center + scan_ranges[i]

            k1_values = np.linspace(k1_min, k1_max, n_steps)

            best_cost = np.inf
            best_k1 = k1_center
            best_sigx = None
            best_sigy = None

            print(f"Scanning {qname}: range [{k1_min:9.4f}, {k1_max:9.4f}] ", end='')

            for k1_test in k1_values:
                line_optics[qname].k1 = float(k1_test)
                cost, sigx, sigy = cost_function(line_optics, Sigma_x0, Sigma_y0, target)

                if cost < best_cost:
                    best_cost = cost
                    best_k1 = k1_test
                    best_sigx = sigx
                    best_sigy = sigy

            # update
            k1[i] = best_k1
            line_optics[qname].k1 = float(best_k1)

            # store results (true initial k1, not k1_center)
            results[qname] = {
                "k1_init": k1_initial[i],
                "k1": best_k1,
                "sigx": best_sigx,
                "sigy": best_sigy,
                "cost": best_cost,
            }

            print(
                f"  → Best {qname:6s} | "
                f"k1={best_k1:11.6f}  "
                f"σx={best_sigx:12.7f}  "
                f"σy={best_sigy:12.7f}  "
                f"cost={best_cost:12.3e}"
            )

        scan_ranges *= shrink_factor
        print(f"\nShrinking scan ranges → {scan_ranges}")

        if np.all(scan_ranges < min_range):
            print("\nStopping: scan ranges below minimum threshold.")
            break

    # final cost
    final_cost, final_sigx, final_sigy = cost_function(line_optics, Sigma_x0, Sigma_y0, target)

    print("\n=== Final Result ===")
    print(f"Final at target:  σx={final_sigx:.7f},  σy={final_sigy:.7f},  cost={final_cost:.3e}\n")

    # header
    print(f"{'Quad':<8} {'k1_init':>12} {'k1_final':>12} {'ratio':>10} "
          f"{'sigx':>12} {'sigy':>12} {'cost':>12}")

    # table rows
    for q in quad_names:
        res = results[q]
        k1_init  = res["k1_init"]
        k1_final = res["k1"]
        ratio    = k1_final / k1_init if k1_init != 0 else float("inf")
        sigx     = res["sigx"]
        sigy     = res["sigy"]
        cost     = res["cost"]

        print(f"{q:<8} {k1_init:12.6f} {k1_final:12.6f} {ratio:10.4f} "
              f"{sigx:12.4g} {sigy:12.4g} {cost:12.4g}")

    return k1, final_sigx, final_sigy, final_cost
