def sigma0_from_beam_xy():
    """
    Return initial sigma matrices and phase-space half-widths
    based on the BEAM card definition.
    """
    sig_x  = 0.15e-2      # m
    sig_xp = 30e-3        # rad
    sig_y  = 0.15e-2      # m
    sig_yp = 120e-3       # rad

    Sigma_x0 = np.array([[sig_x**2,   0.0],
                         [0.0,        sig_xp**2]])

    Sigma_y0 = np.array([[sig_y**2,   0.0],
                         [0.0,        sig_yp**2]])

    phase_space = {
        "x_max":   sig_x,
        "px_max":  sig_xp,
        "y_max":   sig_y,
        "py_max":  sig_yp,
        "zeta_max": 0.0,
        "delta_max": 0.0,
    }

    return Sigma_x0, Sigma_y0, phase_space

def compute_twiss(line, npoints=2000):
    """Compute Twiss and smooth interpolation."""
    init = xt.TwissInit(
        betx=1.0, bety=1.0,
        alfx=-np.tan(np.deg2rad(30.0)),
        alfy=-np.tan(np.deg2rad(120.0)),
        x=0.0, px=0.0, y=0.0, py=0.0,
    )

    tw = line.twiss(method="4d", init=init)
    s_dense = np.linspace(tw.s[0], tw.s[-1], npoints)

    tw_smooth = {
        "s": s_dense,
        "betx": np.interp(s_dense, tw.s, tw.betx),
        "bety": np.interp(s_dense, tw.s, tw.bety),
        "x":    np.interp(s_dense, tw.s, tw.x),
        "y":    np.interp(s_dense, tw.s, tw.y),
        "dx":   np.interp(s_dense, tw.s, tw.dx) if hasattr(tw, "dx") else np.zeros_like(s_dense),
    }

    return tw, tw_smooth

def element_matrix_for_line(line, idx, eps=1e-8):
    """
    Convenience wrapper: get 6x6 matrix for element `idx` in `line`.
    Does NOT change the original element_matrix signature.
    """
    elem = line.elements[idx]
    p_ref = line.particle_ref
    return element_matrix(elem, p_ref, eps=eps)

def get_cumulative_2x2_matrices(line, particle_ref):
    """Return cumulative 2×2 matrices Mx(s), My(s)."""
    Mx = np.eye(2)
    My = np.eye(2)

    Mx_list = []
    My_list = []

    for elem in line.elements:
        M6 = element_matrix(elem, particle_ref)
        Mx_elem = M6[0:2, 0:2]
        My_elem = M6[2:4, 2:4]

        Mx = Mx_elem @ Mx
        My = My_elem @ My

        Mx_list.append(Mx.copy())
        My_list.append(My.copy())

    return Mx_list, My_list

def propagate_sigma(line, tw, emit_x=None, emit_y=None):
    """Propagate geometric sigma using cumulative 2×2 matrices."""
    Sigma_x0, Sigma_y0, _ = sigma0_from_beam_xy()
    Mx_list, My_list = get_cumulative_2x2_matrices(line, line.particle_ref)

    sig_x = []
    sig_y = []

    for Mx, My in zip(Mx_list, My_list):
        Sx = Mx @ Sigma_x0 @ Mx.T
        Sy = My @ Sigma_y0 @ My.T
        sig_x.append(np.sqrt(Sx[0, 0]))
        sig_y.append(np.sqrt(Sy[0, 0]))

    sig_x = np.insert(sig_x, 0, np.sqrt(Sigma_x0[0, 0]))
    sig_y = np.insert(sig_y, 0, np.sqrt(Sigma_y0[0, 0]))

    return np.array(sig_x), np.array(sig_y)

def compute_R_matrix_dict(line):
    import numpy as np

    n_elem = len(line.elements)
    R12 = np.zeros(n_elem)
    R34 = np.zeros(n_elem)

    M_acc = np.eye(4)

    for i in range(n_elem):

        M_elem = element_matrix_for_line(line, i)  # <-- use wrapper
        M_acc = M_elem[0:4, 0:4] @ M_acc

        R12[i] = M_acc[0, 1]
        R34[i] = M_acc[2, 3]

    return {"R12": R12, "R34": R34}

def element_matrix(element, particle_ref, eps=1e-8):
    """
    6x6 finite-difference matrix for a single element.
    Used by get_cumulative_2x2_matrices() and propagate_sigma().
    """

    import numpy as np

    coord_names = ["x", "px", "y", "py", "zeta", "delta"]
    R = np.zeros((6, 6))

    # Convert a particle to a flat 6-vector
    def vec(p):
        return np.array([
            float(p.x),
            float(p.px),
            float(p.y),
            float(p.py),
            float(p.zeta),
            float(p.delta),
        ], dtype=float)

    for j, cname in enumerate(coord_names):

        # fresh reference for each column
        p0 = particle_ref.copy()
        p_plus  = p0.copy()
        p_minus = p0.copy()

        setattr(p_plus,  cname, getattr(p_plus,  cname) + eps)
        setattr(p_minus, cname, getattr(p_minus, cname) - eps)

        # track through this element only
        element.track(p_plus)
        element.track(p_minus)

        # finite difference derivative
        diff = (vec(p_plus) - vec(p_minus)) / (2 * eps)

        # ensure shape is (6,)
        R[:, j] = diff.flatten()

    return R

def make_beam(
    line,
    Sigma_x0, Sigma_y0,
    n_part=1_000_000,
    transverse="gaussian",     # "gaussian" or "flat"
    longitudinal="gaussian",   # "gaussian" or "flat"
    sigma_delta=1e-3,          # width for gaussian delta
    delta_max=3e-3,            # half-width for flat delta
    seed=12345,
    prt=0,
    species='pi+',
):
    import numpy as np
    import xpart as xp
    from muse_tools.particles import get_particle_mass, get_particle_pdg

    rng = np.random.default_rng(seed)

    # ------------------------------------------------------------
    # 1) Transverse distributions
    # ------------------------------------------------------------
    if transverse == "gaussian":
        x  = rng.normal(0.0, np.sqrt(Sigma_x0[0, 0]), size=n_part)
        px = rng.normal(0.0, np.sqrt(Sigma_x0[1, 1]), size=n_part)
        y  = rng.normal(0.0, np.sqrt(Sigma_y0[0, 0]), size=n_part)
        py = rng.normal(0.0, np.sqrt(Sigma_y0[1, 1]), size=n_part)

    elif transverse == "flat":
        x  = rng.uniform(-np.sqrt(3*Sigma_x0[0, 0]),  np.sqrt(3*Sigma_x0[0, 0]),  size=n_part)
        px = rng.uniform(-np.sqrt(3*Sigma_x0[1, 1]),  np.sqrt(3*Sigma_x0[1, 1]),  size=n_part)
        y  = rng.uniform(-np.sqrt(3*Sigma_y0[0, 0]),  np.sqrt(3*Sigma_y0[0, 0]),  size=n_part)
        py = rng.uniform(-np.sqrt(3*Sigma_y0[1, 1]),  np.sqrt(3*Sigma_y0[1, 1]),  size=n_part)

    else:
        raise ValueError("transverse must be 'gaussian' or 'flat'")

    # ------------------------------------------------------------
    # 2) Longitudinal distribution
    # ------------------------------------------------------------
    if longitudinal == "gaussian":
        delta = rng.normal(0.0, sigma_delta, size=n_part)

    elif longitudinal == "flat":
        delta = rng.uniform(-delta_max, delta_max, size=n_part)

    else:
        raise ValueError("longitudinal must be 'gaussian' or 'flat'")

    zeta = np.zeros(n_part)

    # ------------------------------------------------------------
    # 3) Particle species (mass + PDG ID)
    # ------------------------------------------------------------
    mass = get_particle_mass(species)
    pdg  = get_particle_pdg(species)

    # ------------------------------------------------------------
    # 4) Construct particles
    # ------------------------------------------------------------
    particles = xp.Particles(
        mass=mass,
        p0c=line.particle_ref.p0c,   # momentum defined in main
        pdg_id=pdg,
        x=x, px=px,
        y=y, py=py,
        zeta=zeta, delta=delta,
    )

    # ------------------------------------------------------------
    # 5) Optional printing
    # ------------------------------------------------------------
    if prt:
        if transverse == "gaussian":
            print_beam_stats(particles, "Gaussian beam")
        else:
            print_flat_beam_stats(particles, "Flat beam")

    return particles

def print_beam_stats(particles, label=""):
    x  = particles.x
    px = particles.px
    y  = particles.y
    py = particles.py
    d  = particles.delta

    print(f"--- {label} ---")
    print(f"sigma_x  = {np.std(x):.3g}")
    print(f"sigma_px = {np.std(px):.3g}")
    print(f"sigma_y  = {np.std(y):.3g}")
    print(f"sigma_py = {np.std(py):.3g}")
    print(f"sigma_delta = {np.std(d):.3g}")
    print(f"N = {len(x)}")

def print_flat_beam_stats(particles, label=""):
    x  = particles.x
    px = particles.px
    y  = particles.y
    py = particles.py
    d  = particles.delta

    print(f"--- {label} ---")
    print(f"half_x  = {np.max(np.abs(x)):.3g},   sigma_x  = {np.std(x):.3g}")
    print(f"half_px = {np.max(np.abs(px)):.3g},  sigma_px = {np.std(px):.3g}")
    print(f"half_y  = {np.max(np.abs(y)):.3g},   sigma_y  = {np.std(y):.3g}")
    print(f"half_py = {np.max(np.abs(py)):.3g},  sigma_py = {np.std(py):.3g}")
    print(f"half_delta = {np.max(np.abs(d)):.3g}, sigma_delta = {np.std(d):.3g}")
    print(f"N = {len(x)}")
