def store_phase_space(line, particles, select_keys=None, s_storage=None, keep_full=False):
    """
    Store phase space at selected element entrances or user-defined s positions.
    """

    EV_TO_MEV = 1e-6

    part = particles.copy()
    part.at_element[:] = 0
    part.s[:] = 0

    names = line.element_names
    s_start = line.get_s_elements()
    n_elems = len(names)

    storage_points = []

    # --- Only store selected element names ---
    if select_keys is not None:
        for key in select_keys:
            idx = names.index(key)
            storage_points.append((f"{key}", idx))

    # --- Optional: user-defined s positions ---
    if s_storage is not None:
        for s in s_storage:
            idx = np.searchsorted(s_start, s)
            idx = min(max(idx, 0), n_elems - 1)
            label = f"s={s:.3f}" if isinstance(s, (float, int)) else f"s={s}"
            storage_points.append((label, idx))

    storage_points.sort(key=lambda x: x[1])

    # Reference quantities
    p0c = float(part.p0c[0])
    mass0 = float(part.mass0)
    p0c_MeV = p0c * EV_TO_MEV
    mass0_MeV = mass0 * EV_TO_MEV

    snapshots = {}
    current_element = 0

    for label, idx_target in storage_points:

        if idx_target > current_element:
            line.track(part, ele_start=current_element, ele_stop=idx_target)
            current_element = idx_target

        s_here = s_start[idx_target]

        delta = part.delta
        p = p0c * (1.0 + delta)

        Px = part.px * p0c
        Py = part.py * p0c
        Pz = np.sqrt(np.maximum(p**2 - Px**2 - Py**2, 0.0))
        E = np.sqrt(p**2 + mass0**2)

        Px *= EV_TO_MEV
        Py *= EV_TO_MEV
        Pz *= EV_TO_MEV
        p  *= EV_TO_MEV
        E  *= EV_TO_MEV

        n = len(part.x)

        # --- Minimal columns (recommended) ---
        if not keep_full:
            df = pd.DataFrame({
                "x": part.x.copy(),
                "y": part.y.copy(),
                "px_norm": part.px.copy(),
                "py_norm": part.py.copy(),
                "delta": delta.copy(),
                "state": part.state.copy(),
            })

        # --- Full physics columns (old behavior) ---
        else:
            df = pd.DataFrame({
                "particle_id": part.particle_id.copy(),
                "pdg_id": part.pdg_id.copy(),
                "state": part.state.copy(),

                "x": part.x.copy(),
                "y": part.y.copy(),
                "z": part.zeta.copy(),
                "s": np.full(n, s_here),

                "px_norm": part.px.copy(),
                "py_norm": part.py.copy(),
                "delta": delta.copy(),

                "Px": Px,
                "Py": Py,
                "Pz": Pz,
                "P": p,
                "E": E,

                "p0c": np.full(n, p0c_MeV),
                "mass": np.full(n, mass0_MeV),
            })

        snapshots[label] = df

    return snapshots

def plot_one_histogram_for_all_keys(
        snapshots, select_keys, histo_name, extractor, labels,
        bins=50, figsize=(12, 10), alive_only=True, fontsize=6, cmap="viridis",labelrotation=0.):
    """
    Draw one histogram type across all keys in a 3×3 grid.

    extractor(x, y, xp, yp, delta) must return:
        - (X,) for 1D histograms
        - (X, Y) for 2D histograms
    """
    fig, axes = plt.subplots(4, 4, figsize=figsize)
    axes = axes.flatten()
    fig.suptitle(histo_name, fontsize=16)

    for ax, key in zip(axes, select_keys):
        df = snapshots[key]
        if alive_only:
            df = df[df.state == 1]

        x = df.x.values
        y = df.y.values
        xp = df.px_norm.values
        yp = df.py_norm.values
        delta = df.delta.values

        data = extractor(x, y, xp, yp, delta)

        if len(data) == 1:
            X = data[0]

            lo, hi = np.percentile(X, [0.1, 99.9])
            ax.set_xlim(lo, hi)
            ax.hist(X, bins=bins, color="steelblue", alpha=0.8)

            # --- statistics box ---
            stats = (
                f"N={len(X)}\n"
                f"μ={np.mean(X):.2g}\n"
                f"σ={np.std(X):.2g}"
            )
            ax.text(
                0.02, 0.98, stats,
                transform=ax.transAxes,
                ha="left", va="top",
                fontsize=fontsize,
                bbox=dict(facecolor="white", alpha=0.6, edgecolor="none")
            )

            ax.set_xlabel(labels[0])
            ax.set_ylabel("count")
            ax.tick_params(axis='x', labelrotation=labelrotation)
            ax.tick_params(axis='x', labelsize=8)
            ax.tick_params(axis='y', labelsize=8)

        else:
            X, Y = data

            xlo, xhi = np.percentile(X, [0.1, 99.9])
            ylo, yhi = np.percentile(Y, [0.1, 99.9])

            h = ax.hist2d(X, Y, bins=bins, range=[[xlo, xhi], [ylo, yhi]], cmap=cmap)
            fig.colorbar(h[3], ax=ax)

            ax.set_xlim(xlo, xhi)
            ax.set_ylim(ylo, yhi)

            # --- statistics box ---
            rho = np.corrcoef(X, Y)[0, 1]
            stats = (
                f"N={len(X)}\n"
                f"σₓ={np.std(X):.2g}\n"
                f"σᵧ={np.std(Y):.2g}\n"
                f"ρ={rho:.3f}"
            )
            ax.text(
                0.02, 0.98, stats,
                transform=ax.transAxes,
                ha="left", va="top",
                fontsize=fontsize,
                bbox=dict(facecolor="white", alpha=0.6, edgecolor="none")
            )

            ax.set_xlabel(labels[0])
            ax.set_ylabel(labels[1])
            ax.tick_params(axis='x', labelrotation=labelrotation)
            ax.tick_params(axis='x', labelsize=8)
            ax.tick_params(axis='y', labelsize=8)

        ax.set_title(key)

    for ax in axes[len(select_keys):]:
        ax.axis("off")

    fig.tight_layout()
    return fig

def save_all_histograms_to_pdf(
        snapshots, select_keys, histograms,
        pdf_filename="phase_space_histograms.pdf",
        pkl_filename="phase_space_histograms.pkl",
        bins=120, figsize=(12, 10), alive_only=True, cmap="viridis" ):

    all_hist_data = {}   # <-- collect all histogram arrays here

    with PdfPages(pdf_filename) as pdf:
        for name, extractor, labels in histograms:

            # --- generate the figure ---
            fig = plot_one_histogram_for_all_keys(
                snapshots, select_keys, name, extractor, labels,
                bins=bins, figsize=figsize, alive_only=alive_only, cmap=cmap
            )
            pdf.savefig(fig)
            plt.close(fig)

            # --- collect raw histogram data for pkl ---
            per_key_data = {}
            for key in select_keys:
                df = snapshots[key]
                if alive_only:
                    df = df[df.state == 1]

                x = df.x.values
                y = df.y.values
                xp = df.px_norm.values
                yp = df.py_norm.values
                delta = df.delta.values

                data = extractor(x, y, xp, yp, delta)
                per_key_data[key] = data

            all_hist_data[name] = {
                "labels": labels,
                "data": per_key_data,
            }

    # --- write pickle file ---
    with open(pkl_filename, "wb") as f:
        pickle.dump(all_hist_data, f)

    return pdf_filename, pkl_filename

def print_track_stats(particles, label=""):
    alive = np.sum(particles.state == 1)
    lost  = np.sum(particles.state == 0)
    total = len(particles.state)

    if label:
        print(f"--- {label} ---")
    print(f"tracks valid: {alive}")
    print(f"tracks lost:  {lost}")
    print(f"tracks all:   {total}")

def verify_final_solution_optics(line_optics, k1_final, snapshot_keys, Sigma_x0, Sigma_y0):
    """
    Optics-only verification: apply k1 values to line_optics and
    compute sigma_x, sigma_y at each snapshot location using
    matrix-based propagation. No tracking, no histograms.
    """

    quad_names = ["QSL18", "QSL17", "QSL16"]
    for name, k1 in zip(quad_names, k1_final):
        line_optics[name].k1 = float(k1)

    # cumulative matrices
    Mx_list, My_list = get_cumulative_2x2_matrices(
        line_optics, line_optics.particle_ref
    )

    results = {}
    for key in snapshot_keys:
        idx = line_optics.element_names.index(key)

        Mx = Mx_list[idx]
        My = My_list[idx]

        Sigma_x = Mx @ Sigma_x0 @ Mx.T
        Sigma_y = My @ Sigma_y0 @ My.T

        sigx = np.sqrt(Sigma_x[0, 0])
        sigy = np.sqrt(Sigma_y[0, 0])

        results[key] = (sigx, sigy)

    return results

def run_final_verification(
        line_full,
        k1_final,
        Sigma_x0,
        Sigma_y0,
        snapshot_keys,
        histograms,
        pdf_filename,
        pkl_filename,
        n_part=int(1e7),
        seed=1,
        transverse="flat",
        longitudinal="flat",
        sigma_delta=1e-3,
        delta_max=3e-2,
        species=None,
    ):
    """
    Run the final verification for the beamline simulation.

    Parameters
    ----------
    line_full : xtrack.Line
        The full Xsuite line object.
    k1_final : list
        Final quadrupole strengths.
    ... (add other params if you like) ...
    """


    # --- local imports so this can run in main without recompiling the module ---
    '''
    from muse_tools.histograms import (
        store_phase_space,
        save_all_histograms_to_pdf,
    )
    '''
    
    from muse_tools.optics import make_beam  # adjust if your make_beam lives elsewhere

    # 1) High-statistics beam
    particles = make_beam(
        line_full,
        Sigma_x0, Sigma_y0,
        n_part=n_part,
        transverse=transverse,
        longitudinal=longitudinal,
        sigma_delta=sigma_delta,
        delta_max=delta_max,
        seed=seed,
        species=species,
    )

    # 2) Apply final k1 settings
    quad_names = ["QSL18", "QSL17", "QSL16"]
    for name, k1 in zip(quad_names, k1_final):
        line_full[name].k1 = float(k1)

    # 3) Track and store snapshots
    snapshots = store_phase_space(
        line=line_full,
        particles=particles,
        select_keys=snapshot_keys,
        s_storage=None,
        keep_full=False,
    )

    # 4) Save all histograms using the existing list from main
    save_all_histograms_to_pdf(
        snapshots=snapshots,
        select_keys=snapshot_keys,
        histograms=histograms,
        pdf_filename=pdf_filename,
        pkl_filename=pkl_filename,
    )

    return snapshots
