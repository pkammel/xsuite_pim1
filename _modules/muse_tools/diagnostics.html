def plot_beta_and_envelope(line, tw, emit_x=1e-6, emit_y=1e-6, lw=2, figsize=(10, 10)):
    """Plot beta functions and geometric envelope."""
    s = tw.s
    betx = tw.betx
    bety = tw.bety
    dx = tw.dx if hasattr(tw, "dx") else np.zeros_like(s)

    sig_x, sig_y = propagate_sigma(line, tw)

    positions = get_element_positions(line)

    fig, (ax_beta, ax_env) = plt.subplots(2, 1, figsize=(14, 10), sharex=True)

    ax_beta.plot(s, betx, label="βx", color="darkorange", linewidth=lw)
    ax_beta.plot(s, bety, label="βy", color="green", linewidth=lw)
    ax_beta.grid(True)
    ax_beta.set_ylabel("β [m]")

    ax2 = ax_beta.twinx()
    ax2.plot(s, dx, label="Dx", color="steelblue", linewidth=lw)
    ax2.set_ylabel("Dx [m]", color="steelblue")

    ymax = max(betx.max(), bety.max())
    for name, etype, spos, L in positions:
        if etype in ("Quadrupole", "Multipole", "Bend") and L > 0:
            col = element_color(etype)
            ax_beta.add_patch(Rectangle((spos, 0), L, 1,
                                        transform=ax_beta.get_xaxis_transform(),
                                        color=col, alpha=0.2))
            ax_beta.text(spos + L/2, 0.97*ymax, name, rotation=90,
                         fontsize=8, ha="center", va="top")

    ax_env.plot(s, sig_y, label="σy", color="green", linewidth=lw)
    ax_env.plot(s, -sig_x, label="-σx", color="darkorange", linewidth=lw)
    ax_env.grid(True)
    ax_env.set_xlabel("s [m]")
    ax_env.set_ylabel("Beam size σ [m]")

    ax3 = ax_env.twinx()
    ax3.plot(s, dx, label="Dx", color="steelblue", linewidth=lw)
    ax3.set_ylabel("Dx [m]", color="steelblue")

    plt.tight_layout()
    return fig

def plot_overlaid_envelopes(configs, lw=2, apertures=None,figsize=(10,10)):
    """
    Plot overlaid beam envelopes for multiple optics configurations.

    configs: list of dicts with keys:
        - line
        - tw
        - sig_x
        - sig_y
        - label
        - dp_frac
    """

    import numpy as np
    import matplotlib.pyplot as plt
    from matplotlib.patches import Rectangle

    # ------------------------------------------------------------
    # 1. Reference s-grid: Twiss element boundaries
    # ------------------------------------------------------------
    tw0 = configs[0]["tw"]
    s_ref = tw0.s.copy()

    betx = tw0.betx
    bety = tw0.bety
    dx   = tw0.dx if hasattr(tw0, "dx") else np.zeros_like(s_ref)

    # ------------------------------------------------------------
    # 2. Interpolate envelopes for all configs onto Twiss s-grid
    # ------------------------------------------------------------
    for cfg in configs:
        line = cfg["line"]
        s_sigma = np.linspace(0, line.get_length(), len(cfg["sig_x"]))
        cfg["sig_x"] = np.interp(s_ref, s_sigma, cfg["sig_x"])
        cfg["sig_y"] = np.interp(s_ref, s_sigma, cfg["sig_y"])

    # ------------------------------------------------------------
    # 3. Use line from first config for magnet shading
    # ------------------------------------------------------------
    line = configs[0]["line"]
    positions = get_element_positions(line)

    # ------------------------------------------------------------
    # 4. Create figure
    # ------------------------------------------------------------
    fig, (ax_beta, ax_env) = plt.subplots(
        2, 1, figsize=figsize, sharex=True,
        gridspec_kw={'height_ratios': [2, 2]}
    )

    # ------------------------------------------------------------
    # 5. TOP PANEL: βx, βy, Dx with magnet shading
    # ------------------------------------------------------------
    ax_beta.plot(s_ref, betx, label="βx", color="darkorange", linewidth=lw)
    ax_beta.plot(s_ref, bety, label="βy", color="green", linewidth=lw)
    ax_beta.set_ylabel("β [m]")
    ax_beta.grid(True)

    ax2 = ax_beta.twinx()
    ax2.plot(s_ref, dx, label="Dx", color="steelblue", linewidth=lw)
    ax2.set_ylabel("Dx [m]", color="steelblue")
    ax2.tick_params(axis="y", labelcolor="steelblue")

    ymax_beta = max(betx.max(), bety.max())

    for name, etype, spos, L in positions:
        if etype in ("Quadrupole", "Multipole", "Bend") and L > 0:
            col = element_color(etype)
            ax_beta.add_patch(Rectangle(
                (spos, 0), L, 1,
                transform=ax_beta.get_xaxis_transform(),
                color=col, alpha=0.20, zorder=0
            ))
            ax_beta.text(
                spos + L/2, 0.97*ymax_beta, name,
                rotation=90, fontsize=8, ha="center", va="top"
            )

    l1, lab1 = ax_beta.get_legend_handles_labels()
    l2, lab2 = ax2.get_legend_handles_labels()
    ax_beta.legend(l1 + l2, lab1 + lab2, loc="upper right")

    # ------------------------------------------------------------
    # 6. BOTTOM PANEL: Envelopes + σx_total + magnet shading + apertures
    # ------------------------------------------------------------
    ymax_env = 0
    for cfg in configs:
        ymax_env = max(ymax_env, np.max(cfg["sig_y"]))

    for name, etype, spos, L in positions:
        if etype in ("Quadrupole", "Multipole", "Bend") and L > 0:
            col = element_color(etype)
            ax_env.add_patch(Rectangle(
                (spos, -1), L, 2,
                transform=ax_env.get_xaxis_transform(),
                color=col, alpha=0.20, zorder=0
            ))
            ax_env.text(
                spos + L/2, 0.97*ymax_env, name,
                rotation=90, fontsize=8, ha="center", va="top"
            )

    line_styles = ["solid", "dashed", "dotted", "dashdot"]
    for i, cfg in enumerate(configs):
        style = line_styles[i % len(line_styles)]
        dp = cfg.get("dp_frac", 0.0)

        # geometric σy (always)
        ax_env.plot(
            s_ref, cfg["sig_y"],
            label=f"σy geom ({cfg['label']})",
            linewidth=lw,
            linestyle=style,
            color="green",
        )

        # σx total if dp != 0
        if dp != 0.0:
            sig_x_tot = np.sqrt(cfg["sig_x"]**2 + (dx * dp)**2)
            ax_env.plot(
                s_ref, -sig_x_tot,
                label=f"-σx total ({cfg['label']})",
                linewidth=lw,
                linestyle=style,
                color="darkorange",
            )
        else:
            ax_env.plot(
                s_ref, -cfg["sig_x"],
                label=f"-σx geom ({cfg['label']})",
                linewidth=lw,
                linestyle=style,
                color="darkorange",
            )

    # ------------------------------------------------------------
    # 7. Aperture overlays
    # ------------------------------------------------------------
    if apertures is not None:
        for ap in apertures:
            s0 = ap["s"]
            ax_env.axvline(s0, color="black", linestyle="--", alpha=0.4)

            if "half_ap_x" in ap:
                ax_env.hlines(ap["half_ap_x"], s0-0.05, s0+0.05, color="red", linewidth=2)
                ax_env.hlines(-ap["half_ap_x"], s0-0.05, s0+0.05, color="red", linewidth=2)

            if "half_ap_y" in ap:
                ax_env.hlines(ap["half_ap_y"], s0-0.05, s0+0.05, color="blue", linewidth=2)
                ax_env.hlines(-ap["half_ap_y"], s0-0.05, s0+0.05, color="blue", linewidth=2)

            ax_env.text(s0, 1.05*ymax_env, ap["name"], ha="center", va="bottom", fontsize=9)

    # ------------------------------------------------------------
    # 8. Dx twin axis + legends
    # ------------------------------------------------------------
    ax_env.set_xlabel("s [m]")
    ax_env.set_ylabel("Beam size σ [m]")
    ax_env.grid(True)

    ax3 = ax_env.twinx()
    ax3.plot(s_ref, dx, label="Dx", color="steelblue", linewidth=lw)
    ax3.set_ylabel("Dx [m]", color="steelblue")
    ax3.tick_params(axis="y", labelcolor="steelblue")

    l3, lab3 = ax_env.get_legend_handles_labels()
    l4, lab4 = ax3.get_legend_handles_labels()
    ax_env.legend(l3 + l4, lab3 + lab4, loc="lower left")

    plt.tight_layout()
    return fig

def plot_optics_sheet(configs, figsize=(14, 12), lw=2):
    """
    TRANSPORT-style optics sheet with 4 panels:
      1. Beta functions (βx, βy) + Dx
      2. Dispersion-inflated envelopes σx_total and geometric σy
      3. R12, R34 accumulated transport matrix elements
      4. Kx, Ky focusing strengths (piecewise constant)

    Uses the same input structure as plot_overlaid_envelopes(configs).
    """

    import numpy as np
    import matplotlib.pyplot as plt
    from matplotlib.patches import Rectangle

    # ------------------------------------------------------------
    # Extract nominal optics from configs[0]
    # ------------------------------------------------------------
    cfg0 = configs[0]
    line = cfg0["line"]
    tw   = cfg0["tw"]

    s    = tw.s
    betx = tw.betx
    bety = tw.bety
    dx   = tw.dx if hasattr(tw, "dx") else np.zeros_like(s)

    # ------------------------------------------------------------
    # Geometry
    # ------------------------------------------------------------
    names   = line.element_names
    s_start = line.get_s_elements()
    lengths = [getattr(elem, "length", 0.0) for elem in line.elements]

    # ------------------------------------------------------------
    # Compute R-matrix accumulation
    # ------------------------------------------------------------
    R = compute_R_matrix_dict(line)
    R12 = R["R12"]
    R34 = R["R34"]

    # ------------------------------------------------------------
    # Compute Kx, Ky piecewise-constant focusing strengths
    # ------------------------------------------------------------
    Kx = np.zeros_like(s)
    Ky = np.zeros_like(s)

    for i, elem in enumerate(line.elements):
        if hasattr(elem, "k1"):
            Kx[i] =  elem.k1
            Ky[i] = -elem.k1
        else:
            Kx[i] = 0
            Ky[i] = 0

    # ------------------------------------------------------------
    # Figure with 4 stacked panels
    # ------------------------------------------------------------
    fig, axs = plt.subplots(
        4, 1, figsize=figsize, sharex=True,
        gridspec_kw={'height_ratios': [2, 2, 2, 2]}
    )

    ax_beta, ax_env, ax_R, ax_K = axs

    # ------------------------------------------------------------
    # Panel 1: Beta functions + Dx
    # ------------------------------------------------------------
    ax_beta.plot(s, betx, label="βx", color="darkorange", linewidth=lw)
    ax_beta.plot(s, bety, label="βy", color="green", linewidth=lw)
    ax_beta.set_ylabel("β [m]")
    ax_beta.grid(True)

    ax2 = ax_beta.twinx()
    ax2.plot(s, dx, label="Dx", color="steelblue", linewidth=lw)
    ax2.set_ylabel("Dx [m]", color="steelblue")
    ax2.tick_params(axis="y", labelcolor="steelblue")

    ymax = max(betx.max(), bety.max())

    # Magnet shading + names
    for name, s0, L, elem in zip(names, s_start, lengths, line.elements):
        if L <= 0:
            continue

        etype = elem.__class__.__name__
        if "Quadrupole" in etype:
            col = "red"
        elif "Bend" in etype:
            col = "blue"
        elif "Multipole" in etype:
            col = "purple"
        else:
            continue

        ax_beta.add_patch(Rectangle(
            (s0, 0), L, 1,
            transform=ax_beta.get_xaxis_transform(),
            color=col, alpha=0.20, zorder=0
        ))

        ax_beta.text(
            s0 + L/2, 0.97*ymax, name,
            rotation=90, fontsize=8,
            ha="center", va="top"
        )

    # ------------------------------------------------------------
    # Panel 2: σx_total and σy
    # ------------------------------------------------------------
    for cfg in configs:
        dp = cfg["dp_frac"]
        sig_x = cfg["sig_x"]
        sig_y = cfg["sig_y"]

        sig_x_total = np.sqrt(sig_x**2 + (dx * dp)**2)

        ax_env.plot(s, sig_x_total, label=f"σx_total ({cfg['label']})",
                    linewidth=lw, color="darkorange")
        ax_env.plot(s, sig_y, label=f"σy ({cfg['label']})",
                    linewidth=lw, color="green")

    ax_env.set_ylabel("σ [m]")
    ax_env.grid(True)

    # ------------------------------------------------------------
    # Panel 3: R12, R34
    # ------------------------------------------------------------
    ax_R.plot(s, R12, label="R12", color="purple", linewidth=lw)
    ax_R.plot(s, R34, label="R34", color="brown", linewidth=lw)
    ax_R.set_ylabel("R-matrix")
    ax_R.grid(True)

    # ------------------------------------------------------------
    # Panel 4: Kx, Ky
    # ------------------------------------------------------------
    ax_K.step(s, Kx, where="post", label="Kx", color="red", linewidth=lw)
    ax_K.step(s, Ky, where="post", label="Ky", color="blue", linewidth=lw)
    ax_K.set_ylabel("K [1/m]")
    ax_K.set_xlabel("s [m]")
    ax_K.grid(True)

    # ------------------------------------------------------------
    # Legends
    # ------------------------------------------------------------
    ax_beta.legend(loc="upper right")
    ax_env.legend(loc="upper right")
    ax_R.legend(loc="upper right")
    ax_K.legend(loc="upper right")

    plt.tight_layout()
    return fig
